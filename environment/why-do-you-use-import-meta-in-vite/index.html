нҡҢмӮ¬м—җм„ң vue-cliлЎң л§Ңл“Өм–ҙм§„ н”„лЎңм қнҠёлҘј ViteлЎң л§Ҳмқҙк·ёл Ҳмқҙм…ҳмқ„ м§„н–ү мӨ‘мқҙлӢӨ. vue-cliлҠ” 2л…„ м „л¶Җн„° maintenance mode(deprecated)м—җ л“Өм–ҙк°”кі , webpackмқ„ кё°л°ҳмңјлЎң л§Ңл“Өм–ҙмЎҢкё° л•Ңл¬ём—җ лІҲл“Өл§Ғ л°Ҹ л№Ңл“ң н•  л•Ң л¬ҙмІҷ лҠҗлҰ¬лӢӨ. vue-cli нҢЁнӮӨм§Җ лӮҙл¶Җм—җлҠ” eslint, prettierк°Җ vue-cliм—җ л§һкІҢ л¬¶м—¬мһҲм—ҲлҠ”лҚ° мөңмӢ  лІ„м „мқҳ eslint, prettier лІ„м „мқ„ мӮ¬мҡ©н•ҳкі  мӢ¶м–ҙлҸ„ нҳёнҷҳмқҙ лҗҳм§Җ м•Ҡм•ҳлӢӨ. к·ёлһҳм„ң vue-cli лҢҖмӢ  лІҲл“Өлҹ¬лҘј л°”кҝ” к°ңл°ң нҷҳкІҪмқ„ мөңмӢ нҷ”н•ҳкё°лЎң н–ҲлӢӨ.

мҳҲм „л¶Җн„° мӮ¬мҡ©н•ҙ ліҙкі  мӢ¶м—ҲлҚҳ лІҲл“Өлҹ¬лҠ” ViteлӢӨ. Viteмқҳ к°ҖмһҘ нҒ° нҠ№м§•мқҖ devServerм—җм„ң мҶҢмҠӨмҪ”л“ңмҷҖ л””нҺңлҚҳмӢңлҘј л‘җ к°Җм§Җ м№ҙн…Ңкі лҰ¬лЎң лӮҳлҲ„м–ҙ л№Ңл“ңн•ҳкі , ESM нҷҳкІҪмқё лёҢлқјмҡ°м Җм—җ л№ лҘҙкІҢ м ңкіөн•ңлӢӨлҠ” м җмқҙлӢӨ. devServerк°Җ cold-start лҗ  л•Ң нҳ„мһ¬ нҺҳмқҙм§Җм—җм„ң н•„мҡ”н•ң м •ліҙл§Ң м•„мЈј л№ лҘҙкІҢ мӢӨн–үмӢңнӮ¬ мҲҳ мһҲлӢӨ. мқҙ л§җмқҖ, мҪ”л“ң лІ мқҙмҠӨ лҳҗлҠ” лқјмқҙлёҢлҹ¬лҰ¬(мӮ¬мҡ©н•ҳкі  мһҲлҠ” node_modules л””нҺңлҚҳмӢң)мқҳ лӘЁл“  нҢҢмқјмқ„ лІҲл“Өл§Ғ н•ҳм—¬ м ңкіөн•ҳлҠ” кІғмқҙ м•„лӢҲлқј, нҳ„мһ¬ н•„мҡ”н•ң лӘЁл“Ҳл§Ң кіЁлқјлӮҙм–ҙ л№ лҘҙкІҢ м ңкіөн•ңлӢӨлҠ” лң»мқҙлӢӨ. Vite к°Җмқҙл“ңм—җм„ң к°ҖмһҘ лЁјм Җ мқҙм•јкё°н•ҳкі  мһҲлҠ” кІғмқҙ ESMмқ„ нҶөн•ң devServer start мҶҚлҸ„мқҙлӢӨ.

мӮ¬м „ лІҲл“Өл§ҒмқҖ, esBuildлҘј нҶөн•ҙ л””нҺңлҚҳмӢңлҘј л№Ңл“ңн•ҳкі  ліҖкІҪмқҙ мһҰмқҖ мҶҢмҠӨмҪ”л“ңлҠ” ESMмңјлЎң м ңкіөн•ңлӢӨ. кё°мЎҙм—җ Webpackкіј к°ҷмқҖ лІҲл“Өлҹ¬лҠ” лӘЁл“  нҢҢмқјмқ„ л№Ңл“ң л°Ҹ лІҲл“Өл§Ғ н•ҙм•јл§Ң лёҢлқјмҡ°м Җм—җм„ң кө¬лҸҷмқҙ к°ҖлҠҘн–ҲлӢӨ. ViteлҠ” мҶҢмҠӨмҪ”л“ңлҘј ESMмңјлЎң м ңкіөн•ҳкё° л•Ңл¬ём—җ лёҢлқјмҡ°м Җм—җм„ң лҸҷм Ғ importлҘј нҶөн•ҙ мӣҗн•ҳлҠ” JavaScript нҢҢмқјл§Ң мӢӨн–үмӢңнӮ¬ мҲҳ мһҲлӢӨ. мқҙ л§Өм»ӨлӢҲмҰҳм—җ л”°лқј Webpackмқҳ Node.jsм—җм„ңл§Ң к°ҖлҠҘн–ҲлҚҳ кё°лҠҘмқҙ лёҢлқјмҡ°м Җм—җм„ң лҸҷмһ‘н•ҳкё° мң„н•ҙ м–ҙл–»кІҢ Viteк°Җ н•ҙкІ°н–ҲлҠ”м§ҖлҘј м•Ңм•„ліј кІғмқҙлӢӨ.


Node.js нҷҳкІҪ ліҖмҲҳ

process.env

лҢҖл¶Җл¶„мқҳ bundlerлҠ” Node.js нҷҳкІҪ мң„м—җм„ң лҸҷмһ‘н•ңлӢӨ. Node.jsм—җм„ңлҠ” process лқјлҠ” нҠ№лі„н•ң мқҙлҰ„мқҳ мҳҲм•Ҫм–ҙлҘј мӮ¬мҡ©н•ҙ нҷҳкІҪ ліҖмҲҳм—җ м ‘к·јн•  мҲҳ мһҲлӢӨ. process мҳҲм•Ҫм–ҙлҠ” нҳ„мһ¬ мӢӨн–үлҗҳкі  мһҲлҠ” Node.js н”„лЎңм„ёмҠӨм—җ лҢҖн•ң м •ліҙлҘј к°Җм§Җкі  мһҲкі  мқҙлҘј м ңм–ҙн•  мҲҳ мһҲлҠ” к°қмІҙлӢӨ.

лҢҖл¶Җл¶„мқҳ JavaScript н”„л Ҳмһ„мӣҢнҒ¬м—җм„ңлҠ” process.env ліҖмҲҳлҘј нҶөн•ҙ нҷҳкІҪ ліҖмҲҳм—җ м ‘к·јн•ңлӢӨ. process.envлҠ” лҹ°нғҖмһ„м—җ лҸҷм ҒмңјлЎң ліҖмҲҳлҘј м¶”к°Җ л°Ҹ мҲҳм •н•  мҲҳ мһҲлӢӨ.

console.log(process.env.TEST); // undefined
process.env.TEST = "test!";
console.log(process.env.TEST); // test!

мң„ мҪ”л“ңлҘј Node.jsмқҳ node лӘ…л №м–ҙлЎң мӢӨн–үн•ҳл©ҙ мЈјм„қкіј к°ҷмқҖ кІ°кіјлҘј м–»мқ„ мҲҳ мһҲлӢӨ. н”„л Ҳмһ„мӣҢнҒ¬м—җм„ңлҠ” .env нҢҢмқјмқ„ нҶөн•ҙ process.env ліҖмҲҳм—җ к°ңл°ңмһҗк°Җ м„Өм •н•ң ліҖмҲҳлҘј process.envм—җ мЈјмһ…мӢңмјңмЈјлҠ”лҚ°, мЈјлЎң dotenv лқјмқҙлёҢлҹ¬лҰ¬лҘј нҶөн•ҙ мЈјмһ…н•ңлӢӨ.

dotenv

лӘЁлҚҳ JavaScript н”„л Ҳмһ„мӣҢнҒ¬м—җм„ң dotenv лқјлҠ” лқјмқҙлёҢлҹ¬лҰ¬лҘј лӮҙл¶Җм ҒмңјлЎң лһҳн•‘н•ҙм„ң мӮ¬мҡ©н•ҳкі  мһҲлӢӨ. dotenv лқјмқҙлёҢлҹ¬лҰ¬лҠ” node.js нҷҳкІҪ л°Ҹ ESM нҷҳкІҪм—җм„ң .env нҢҢмқјм—җ мһҲлҠ” нҷҳкІҪ ліҖмҲҳлҘј лҹ°нғҖмһ„ нҷҳкІҪмқҳ JavaScriptлЎң л“Өкі  мҳ¬ мҲҳ мһҲлӢӨ.

лқјмқҙлёҢлҹ¬лҰ¬лҘј к°ҖліҚкІҢ л¶„м„қн•ҙ ліҙмһҗ.

$ npm init -y
$ npm i -D dotenv

лӘ…л №м–ҙлЎң npm нҷҳкІҪмқ„ мғқм„ұн•ңлӢӨ. package.json нҢҢмқјмқҙ мғқм„ұлҗң нӣ„, dotenv лқјмқҙлёҢлҹ¬лҰ¬лҘј лӢӨмҡҙлЎңл“ң л°ӣм•ҳлӢӨ.

н”„лЎңм қнҠё rootм—җ .env нҢҢмқјмқ„ мғқм„ұн•ҳкі  н…ҢмҠӨнҠё к°’мқ„ л„ЈлҠ”лӢӨ.

# .env file
TEST=ENV_TEST!

main.js нҢҢмқјмқ„ мғқм„ұн•ҳкі , dotenv лқјмқҙлёҢлҹ¬лҰ¬лҘј нҶөн•ҙ .env нҢҢмқјмқҳ нҷҳкІҪ ліҖмҲҳ к°’мқ„ к°Җм ёмҷҖліҙмһҗ.

// src/main.js

require("dotenv").config();
console.log(process.env.TEST);  // ENV_TEST!

мқҙм ң node лӘ…л №м–ҙлЎң Node.js нҷҳкІҪм—җм„ң мқҙлҘј мӢӨн–үмӢңмјңліҙмһҗ.

$ node src/main.js

=> ENV_TEST!

м–ҙл–»кІҢ мқҙлҹ° кІ°кіјк°Җ лӮҳмҳ¬к№Ң? dotenv лқјмқҙлёҢлҹ¬лҰ¬лҘј мӮҙм§қ к№Ңліҙмһҗ. мӢңмһ‘мқҖ require("dotenv").config(); мӨ„мқҙлӢӨ. config л©”м„ңл“ңлҘј мӢӨн–үн•ңлӢӨ.

function config (options) {
  if (_dotenvKey(options).length === 0) {
    return DotenvModule.configDotenv(options);
  }
  // ...
}

configDotenv() н•ЁмҲҳлҘј мӢӨн–үн•ңлӢӨ. lib/main.jsмқҳ configDotenv н•ЁмҲҳ. .env нҢҢмқјмқҳ кІҪлЎңлҘј мқҪкі  нҢҢмӢұ н•ҙ process нҷҳкІҪмқ„ нҢҢмӢұ лҗң к°’мңјлЎң мұ„мҡ°кі  л°ҳнҷҳн•ңлӢӨ.

function configDotenv (options) {
  let dotenvPath = path.resolve(process.cwd(), '.env');
  // ...
  const parsed = DotenvModule.parse(fs.readFileSync(dotenvPath, { encoding }));

  let processEnv = process.env;
  if (options && options.processEnv != null) {
    processEnv = options.processEnv;
  }

  DotenvModule.populate(processEnv, parsed, options);
  return { parsed };
}

    • process.cwd() : node лӘ…л №мқ„ нҳём¶ңн•ң л””л үнҶ лҰ¬мқҳ м ҲлҢҖ кІҪлЎңлӢӨ. /Users/pozafly/Documents/dev/play-ground/dotenv-exam
    • path.resolve() : кІҪлЎңлҘј л¬¶м–ҙ мғҲлЎңмҡҙ кІҪлЎңлҘј л°ҳнҷҳн•ңлӢӨ. /Users/pozafly/Documents/dev/play-ground/dotenv-exam/.env

  2. parsed ліҖмҲҳм—җ .env нҢҢмқјмқ„ нҢҢмӢұ н•ҙ лӢҙлҠ”лӢӨ.

    • fs.readFileSync() : мқёмһҗлЎң л“Өм–ҙмҳӨлҠ” м ҲлҢҖ кІҪлЎңлЎң нҢҢмқј text к°’мқ„ мқҪм–ҙмҳЁлӢӨ. л”°лқјм„ң, dotenvPathмқҳ к°’мқ„ мқҪм–ҙмҳЁлӢӨ.

    • DotenvModule.parse() : нҢҢмӢұ н•ң text к°’мқ„ JavaScript objectлЎң л§Ңл“Өм–ҙм„ң return н•ңлӢӨ.

    • parsed: { TEST: 'ENV_TEST!' } к°’мқҙ лӢҙкІЁмһҲлӢӨ.

  3. populate(processEnv, parsed, options) мӢӨн–үн•ңлӢӨ. populateлҠ” вҖҳлҚ§л¶ҷмқҙлӢӨвҖҷлқјлҠ” лң»мқ„ к°Җм§Җкі  мһҲлӢӨ.

function populate (processEnv, parsed, options = {}) {
  // Set process.env
  for (const key of Object.keys(parsed)) {
    // ...
    processEnv[key] = parsed[key];
  }
}

мң„м—җм„ң processEnv = process.env лЎң л ҲнҚјлҹ°мҠӨ мЈјмҶҢлҘј л¬¶м–ҙмЈјм—Ҳкё° л•Ңл¬ём—җ keyлЎң мҰүмӢң н• лӢ№н•ңлӢӨ. лӢӨмӢң мҡ°лҰ¬к°Җ мһ‘м„ұн–ҲлҚҳ main.js кө¬л¬ёмқ„ ліҙмһҗ.

require('dotenv').config();
console.log(process.env.TEST);  // ENV_TEST!

м •лҰ¬н•ҳл©ҙ, dotenv лқјмқҙлёҢлҹ¬лҰ¬лҠ” Node.js нҷҳкІҪм—җм„ң .env лҘј нҢҢмӢұ н•ҙ text к°’мқ„ нҶөн•ҙ к°қмІҙлҘј мғқм„ұн•ҳкі , process.envм—җ к°’мқ„ м¶”к°Җн•ҙ Node.js нҷҳкІҪмқҳ м „м—ӯм—җ л…ём¶ңлҗңлӢӨ. JavaScript н”„л Ҳмһ„мӣҢнҒ¬м—җм„ңлҠ” ліҙнҶө Node.jsлҘј н•ң лІҲл§Ң мӢӨн–үн•ҳкё° л•Ңл¬ём—җ .env нҢҢмқјмқҳ нҷҳкІҪ ліҖмҲҳ к°’мқҙ ліҖкІҪлҗҳл©ҙ devServerлҘј к»җлӢӨ мјңмЈјм–ҙм•ј к°’мқ„ ліҖкІҪлҗң к°’мңјлЎң л°ҳмҳҒн•  мҲҳ мһҲлҠ” мқҙмң мқҙлӢӨ.

ViteлҠ” л°”лЎң л°ҳмҳҒмқҙ лҗҳлҠ”лҚ° devServerлҘј мҲңк°„м ҒмңјлЎң лӢӨмӢң мӢңмһ‘мӢңмјңмӨҖлӢӨ. .env нҢҢмқјм—җм„ң ліҖмҲҳлҘј ліҖкІҪн•ҳл©ҙ н„°лҜёл„җ logм—җ server restarted.лқјкі  м•Ңл ӨмӨҖлӢӨ. ViteлҠ” WebpackмІҳлҹј лӘЁл“  нҢҢмқјмқ„ лІҲл“Өл§Ғ н•ҳм§Җ м•Ҡкё° л•Ңл¬ём—җ л¬ҙмІҷ л№ лҘҙкё°м—җ к°ҖлҠҘн•ҳлӢӨ.

dotenvм—җм„ң ESM нҷҳкІҪ лҳҗн•ң м§Җмӣҗн•ңлӢӨ.

// package.json
{
  "type": "module",
}

мң„мҷҖ к°ҷмқҙ package.json нҢҢмқјм—җ typeмқ„ moduleлЎң л„Јм–ҙмЈјл©ҙ Node.jsм—җм„ң ESM мҪ”л“ңлҘј мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

вҖ» ViteлЎң л§Ҳмқҙк·ёл Ҳмқҙм…ҳмқ„ м§„н–үн•  л•ҢлҸ„, package.jsonм—җ мң„мҷҖ к°ҷмқҙ "type": "module" мқ„ л„Јм–ҙмЈјм–ҙм•ј н•ҳлҠ”лҚ° ViteлҠ” кё°ліём ҒмңјлЎң ESM л°©мӢқмңјлЎң лҸҷмһ‘н•ңлӢӨ.

м°ёкі лЎң, main.js нҢҢмқјмқ„, main.mjs лқјлҠ” нҷ•мһҘмһҗлҘј л¶ҷм—¬мЈјл©ҙ Node.jsм—җм„ң м•Ңм•„м„ң ESMмңјлЎң лҸҷмһ‘н•ҳкІҢ лҗҳкё° л•Ңл¬ём—җ package.jsonм—җ л”°лЎң л¶ҷм—¬мӨ„ н•„мҡ”лҠ” м—ҶлӢӨ.

vue-cliлҠ” @vue/cli-services нҢЁнӮӨм§Җ лӮҙм—җм„ң мӮ¬мҡ©н•ҳкі  мһҲкі , CRA к°ҷмқҖ кІҪмҡ° react-scripts лқјмқҙлёҢлҹ¬лҰ¬ лӮҙл¶Җ package.jsonм—җ dotenv мқ„ мӮ¬мҡ©н•ҳкі  мһҲмңјл©° Next.jsлҸ„ лӮҙл¶Җм ҒмңјлЎң dotenv лқјмқҙлёҢлҹ¬лҰ¬лҘј мқҙмҡ©н•ҙ .env нҢҢмқјмқ„ к°Җм ёмҳӨкі  мһҲлӢӨ.

вҖ» вҖ”env-file

node.js 20 лІ„м „л¶Җн„°лҠ” --env-file=config лҘј нҶөн•ҙ .env нҢҢмқјмқ„ dotenv лқјмқҙлёҢлҹ¬лҰ¬ м—Ҷмқҙ мӮ¬мҡ©н•  мҲҳ мһҲлҸ„лЎқ к°ңл°ң мӨ‘мқҙлӢӨ. мӢӨм ңлЎң н„°лҜёл„җм—җм„ң node лӘ…л №м–ҙлЎң .env нҢҢмқјмқ„ м§Җм • н•ҙ мӨ„ мҲҳ мһҲлӢӨ. м•„м§ҒмқҖ мӢӨн—ҳмҡ©.

н•ҳм§Җл§Ң, .env нҢҢмқјл§Ң мһҲлҠ” кІғмқҙ м•„лӢҲлқј л°°нҸ¬ нҷҳкІҪм—җ л”°лқј нҷҳкІҪ ліҖмҲҳмқҳ к°’мқҙ лӢ¬лқјм ём•ј нҷҳкІҪ ліҖмҲҳлҘј мӮ¬мҡ©н•ҳлҠ” мқҳлҜёк°Җ мһҲлӢӨ. мҰү, .env.development, .env.production л“ұмқҳ нҷҳкІҪ ліҖмҲҳ нҢҢмқјмқҙ м¶”к°ҖлЎң н•„мҡ”н•ҳкі  modeм—җ л”°лқј м§Җм •лҗң к°’мқҙ process.env к°’м—җ н• лӢ№лҗҳм–ҙ build лҗҳм–ҙм•ј н•ңлӢӨ.

import dotenv from 'dotenv-defaults';

class Dotenv {
  // 1.
  constructor (config = {}) {
    this.config = Object.assign({}, {
      path: './.env',
      prefix: 'process.env.'
    }, config);
  }
  
  // 3.
  getEnvs () {
    const { path } = this.config;
    const env = dotenv.parse(this.loadFile({
      file: path,
    }));
    return {
      env
    };
  }
  
  gatherVariables () {
    const { env } = this.getEnvs(); // 2.
    const vars = Object.assign({}, process.env); // 4.

    Object.keys(env).forEach(key => {
      const value = Object.prototype.hasOwnProperty.call(vars, key) ? vars[key] : env[key];
      vars[key] = value; // 5.
    })

    return vars;
  }
}

к°„м¶”лҰ° мҪ”л“ңл§Ң л“Өкі  мҷ”лӢӨ.

  1. мқёмҠӨн„ҙмҠӨлҘј мғқм„ұн•  л•Ң, л„Јм–ҙмӨҖ pathлҘј мғқм„ұмһҗм—җм„ң м„Өм •н•ңлӢӨ.
  2. getEnvs() л©”м„ңл“ң мӢӨн–ү.
  3. configм—җ лӢҙкёҙ pathлҘј к°Җм ёмҷҖ .env* нҢҢмқјмқ„ к°Җм ёмҳӨкі , dotenv лқјмқҙлёҢлҹ¬лҰ¬м—җ parse л©”м„ңл“ңлҘј мӮ¬мҡ©н•ҙ env к°қмІҙлЎң л§Ңл“Өм–ҙ return.
  4. vars м—җ мғҲлЎңмҡҙ к°қмІҙм—җлӢӨк°Җ process.env к°’мқ„ ліөмӮ¬н•ңлӢӨ.
  5. vars[key] = value лҘј нҶөн•ҙ нҢҢмӢұ н•ң к°’мқ„ м§‘м–ҙл„Јм–ҙ return.

мҰү, dotenv-webpackмқҖ лӢЁмҲңнһҲ .env к°’мқ„ pathлЎң л°ӣм•„ нҢҢмӢұ н•ңлӢӨ.

ViteлҠ” нҷҳкІҪ ліҖмҲҳк°Җ м •мқҳлҗң нҢҢмқј(.env)м—җм„ң м—¬нғҖ лІҲл“Өлҹ¬мҷҖ л§Ҳм°¬к°Җм§ҖлЎң м–ҙн”ҢлҰ¬мјҖмқҙм…ҳм—җ н•„мҡ”н•ң ліҖмҲҳлҘј к°Җм ёмҷҖ мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ. import.meta

import.metaлқјлҠ” JavaScript л¬ёлІ•мқ„ м•Ңм•„ліҙл©° MDN л¬ём„ңм—җ м•„м§Ғ м•„л¬ҙлҸ„ лІҲм—ӯн•ҳм§Җ м•Ҡм•„ мқҙлІҲ кё°нҡҢм—җ н•Ёк»ҳ лІҲм—ӯн•ҙ ліҙм•ҳлӢӨ. import.meta лҠ” ES Moduleмқҙ нғ„мғқн•ҳл©ҙм„ң лӘЁл“Ҳм—җ лҢҖн•ң л©”нғҖ лҚ°мқҙн„°(л¶ҖмҲҳ лҚ°мқҙн„°)лҘј м ҖмһҘн•ҳкё° мң„н•ң к°қмІҙлӢӨ.

м—¬кё°м„ң л§җн•ҳлҠ” л©”нғҖ лҚ°мқҙн„°лһҖ, мқҙлҜём§Җ нҢҢмқјмқ„ мғқк°Ғн•ҳл©ҙ мЎ°кёҲ лҚ” мқҙн•ҙн•ҳкё° мүҪлӢӨ. нңҙлҢҖнҸ°мңјлЎң мӮ¬м§„мқ„ м°Қмңјл©ҙ мӮ¬м§„мқҳ л©”нғҖ лҚ°мқҙн„°м—җ вҖҳлӮ м§ңвҖҷ, вҖҳмӮ¬м§„ мҡ©лҹүвҖҷ, вҖҳл””л°”мқҙмҠӨ м •ліҙвҖҷ, вҖҳмң„м№ҳ лҚ°мқҙн„°вҖҷ л“ұмқҳ л©”нғҖ лҚ°мқҙн„°к°Җ лӢҙкё°кІҢ лҗңлӢӨ. import.meta лҠ” вҖҳлӘЁл“ҲвҖҷ мһҗмІҙм—җ лҢҖн•ң л©”нғҖ лҚ°мқҙн„°лҘј лӢҙкі  мһҲлӢӨ.

import.metaлҠ” лҢҖн‘ңм ҒмңјлЎң лӘЁл“Ҳ л©”нғҖ лҚ°мқҙн„°мқё кІҪлЎң(URL)лҘј лӢҙкі  мһҲлҠ”лҚ° мқҙлҠ” нҳёмҠӨнҠё нҷҳкІҪ(лёҢлқјмҡ°м Җ нҳ№мқҖ Node.js)м—җ л”°лқј м ҖмһҘлҗҳлҠ” лӮҙмҡ©мқҙ лӢ¬лқјм§„лӢӨ. лЁјм Җ Node.jsлҘј нҶөн•ҙ import.metaк°Җ к°Җм§Җкі  мһҲлҠ” м •ліҙлҘј м—ҙм–ҙліҙмһҗ.

HMR(Hot Module Replacement)лҠ” лІҲл“Өлҹ¬ нҷҳкІҪм—җм„ң к°ңл°ңн•  л•Ң мЈјлЎң мӮ¬мҡ©н•ҳлҠ”лҚ°, мҶҢмҠӨмҪ”л“ңлҘј ліҖкІҪн•ҳл©ҙ devServerлҘј мғҲлЎң к»җлӢӨ нӮӨм§Җ м•Ҡкі  лёҢлқјмҡ°м ҖлҘј мғҲлЎңкі м№Ё н•ҳм§Җ м•Ҡм•„лҸ„ лӘЁл“Ҳмқ„ көҗмІҙн•ҙ ліҖкІҪлҗң л¶Җл¶„мқ„ мҰүмӢң ліј мҲҳ мһҲкІҢ л§Ңл“Өм–ҙмЈјлҠ” кё°лҠҘмқҙлӢӨ.

Viteм—җм„ңлҠ” HMR кё°лҠҘм—җ import.meta лҘј мӮ¬мҡ©н•ҳкі  мһҲлӢӨ. лёҢлқјмҡ°м Җм—җ лЎңл“ң лҗң Sources нғӯмқҳ мҶҢмҠӨ л§өм—җ breakpointлҘј м°Қм–ҙ ліҙл©ҙм„ң л”°лқјк°Җліҙмһҗ.
